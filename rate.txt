        function toggleMenu() {
            const menu = document.querySelector('.nav-overlay');
            gsap.to(menu, {right: menu.style.right === '0px'? (window.innerWidth < 768? '-70%': '-300px'): '0px', duration: 0.6, ease: "easeInOutCubic"});
        }

        document.addEventListener('click', function(e) {
            const menu = document.querySelector('.nav-overlay');
            if (!menu.contains(e.target) &&!e.target.closest('.menu-btn')) {
                gsap.to(menu, {right: (window.innerWidth < 768? '-70%': '-300px'), duration: 0.5, ease: "easeInOutCubic"});
            }

        const ratingForm = document.getElementById('ratingForm');
        const ratingMessage = document.getElementById('ratingMessage');
        const fullNameInput = document.getElementById('fullName');
        const discordIdInput = document.getElementById('discordId');
        const driverNameInput = document.getElementById('driverName');
        const starRatingInputs = document.querySelectorAll('input[name="driverRating"]'); // Select all star radio buttons
        const commentsInput = document.getElementById('comments');


        const fullNameError = document.getElementById('fullNameError');
        const discordIdError = document.getElementById('discordIdError');
        const driverNameError = document.getElementById('driverNameError');
        const driverRatingError = document.getElementById('driverRatingError');


        ratingForm.addEventListener('submit', function(event) {
            event.preventDefault();

            let isValid = true;
            let starRatingSelected = false;


            fullNameError.classList.remove('show');
            discordIdError.classList.remove('show');
            driverNameError.classList.remove('show');
            driverRatingError.classList.remove('show');


            if (fullNameInput.value.trim() === '') {
                fullNameError.classList.add('show');
                isValid = false;
            }

            if (discordIdInput.value.trim() === '') {
                discordIdError.classList.add('show');
                isValid = false;
            }

            if (driverNameInput.value.trim() === '') {
                driverNameError.classList.add('show');
                isValid = false;
            }


            for (const starInput of starRatingInputs) {
                if (starInput.checked) {
                    starRatingSelected = true;
                    break;
                }
            }
            if (!starRatingSelected) {
                driverRatingError.classList.add('show');
                isValid = false;
            }


            if (isValid) {
                ratingMessage.classList.add('show');

                ratingForm.reset();
                starRatingInputs.forEach(input => input.checked = false); // Uncheck all star radio buttons


                setTimeout(() => {
                    ratingMessage.classList.remove('show');
                }, 3000);
            }
        });
