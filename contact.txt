    <script>
        AOS.init({
            duration: 1200,
            once: true,
        });

        // GSAP Hero Headline Animation
        gsap.fromTo('.contact-headline', {opacity: 0, y: 20}, {opacity: 1, y: 0, duration: 1.5, ease: "power2.out", delay: 0.5});

        // GSAP for Section Animations on Scroll
        const sections = gsap.utils.toArray('.section');
        sections.forEach(section => {
            gsap.fromTo(section, {opacity: 0, y: 30}, {
                opacity: 1,
                y: 0,
                duration: 1,
                ease: "power2.out",
                scrollTrigger: {
                    trigger: section,
                    start: "top 80%",
                    once: true,
                }
            });
        });


        function toggleMenu() {
            const menu = document.querySelector('.nav-overlay');
            gsap.to(menu, {right: menu.style.right === '0px'? (window.innerWidth < 768? '-70%': '-300px'): '0px', duration: 0.6, ease: "easeInOutCubic"});
        }

        document.addEventListener('click', function(e) {
            const menu = document.querySelector('.nav-overlay');
            if (!menu.contains(e.target) &&!e.target.closest('.menu-btn')) {
                gsap.to(menu, {right: (window.innerWidth < 768? '-70%': '-300px'), duration: 0.5, ease: "easeInOutCubic"});
            }
        });
    </script>
