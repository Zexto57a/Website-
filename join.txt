        const experienceNoInput = document.getElementById('experienceNo');
        const joinReasonInput = document.getElementById('joinReason');
        const wordCountDisplay = document.getElementById('wordCount');


        const fullNameError = document.getElementById('fullNameError');
        const discordIdError = document.getElementById('discordIdError');
        const statsUrlError = document.getElementById('statsUrlError');
        const taxiExperienceError = document.getElementById('taxiExperienceError');
        const joinReasonError = document.getElementById('joinReasonError');


        joinReasonInput.addEventListener('input', function() {
            const words = this.value.trim().split(/\s+/).filter(word => word !== '').length;
            wordCountDisplay.textContent = `${words} words`;
        });


        applicationForm.addEventListener('submit', function(event) {
            event.preventDefault();

            let isValid = true;
            let taxiExperienceSelected = false;


            fullNameError.classList.remove('show');
            discordIdError.classList.remove('show');
            statsUrlError.classList.remove('show');
            taxiExperienceError.classList.remove('show');
            joinReasonError.classList.remove('show');


            if (fullNameInput.value.trim() === '') {
                fullNameError.classList.add('show');
                isValid = false;
            }

            if (discordIdInput.value.trim() === '') {
                discordIdError.classList.add('show');
                isValid = false;
            }

            if (statsUrlInput.value.trim() === '' || !/^(ftp|http|https):\/\/[^ "]+$/.test(statsUrlInput.value)) {
                statsUrlError.classList.add('show');
                isValid = false;
            }


            if (!experienceYesInput.checked && !experienceNoInput.checked) {
                taxiExperienceError.classList.add('show');
                isValid = false;
            } else {
                taxiExperienceSelected = true;
            }


            const words = joinReasonInput.value.trim().split(/\s+/).filter(word => word !== '').length;
            if (words < 150) {
                joinReasonError.classList.add('show');
                isValid = false;
            }


            if (isValid) {
                applicationMessage.classList.add('show');

                applicationForm.reset();
                wordCountDisplay.textContent = '0 words';
                experienceYesInput.checked = false;
                experienceNoInput.checked = false;


                setTimeout(() => {
                    applicationMessage.classList.remove('show');
                }, 3000);
            }
        });
    </script>
</body>
</html>
