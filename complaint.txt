        function toggleMenu() {
            const menu = document.querySelector('.nav-overlay');
            gsap.to(menu, {right: menu.style.right === '0px'? (window.innerWidth < 768? '-70%': '-300px'): '0px', duration: 0.6, ease: "easeInOutCubic"});
        }

        document.addEventListener('click', function(e) {
            const menu = document.querySelector('.nav-overlay');
            if (!menu.contains(e.target) &&!e.target.closest('.menu-btn')) {
                gsap.to(menu, {right: (window.innerWidth < 768? '-70%': '-300px'), duration: 0.5, ease: "easeInOutCubic"});
            }
        });


        const complaintForm = document.getElementById('complaintForm');
        const complaintMessage = document.getElementById('complaintMessage');
        const fullNameInput = document.getElementById('fullName');
        const discordIdInput = document.getElementById('discordId');
        const incidentDateInput = document.getElementById('incidentDate');
        const complaintNatureSelect = document.getElementById('complaintNature');
        const complaintDescriptionTextarea = document.getElementById('complaintDescription');


        const fullNameError = document.getElementById('fullNameError');
        const discordIdError = document.getElementById('discordIdError');
        const incidentDateError = document.getElementById('incidentDateError');
        const complaintNatureError = document.getElementById('complaintNatureError');
        const complaintDescriptionError = document.getElementById('complaintDescriptionError');


        complaintForm.addEventListener('submit', function(event) {
            event.preventDefault();

            let isValid = true;


            fullNameError.classList.remove('show');
            discordIdError.classList.remove('show');
            incidentDateError.classList.remove('show');
            complaintNatureError.classList.remove('show');
            complaintDescriptionError.classList.remove('show');


            if (fullNameInput.value.trim() === '') {
                fullNameError.classList.add('show');
                isValid = false;
            }

            if (discordIdInput.value.trim() === '') {
                discordIdError.classList.add('show');
                isValid = false;
            }


            if (incidentDateInput.value.trim() === '') {
                incidentDateError.classList.add('show');
                isValid = false;
            }

            if (complaintNatureSelect.value === '') {
                complaintNatureError.classList.add('show');
                isValid = false;
            }

            if (complaintDescriptionTextarea.value.trim() === '') {
                complaintDescriptionError.classList.add('show');
                isValid = false;
            }


            if (isValid) {
                complaintMessage.classList.add('show');
                complaintForm.reset();


                setTimeout(() => {
                    complaintMessage.classList.remove('show');
                }, 3000);
            }
        });
    </script>
